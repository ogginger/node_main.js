//testsuite.js: Testing Logic.

define([
  "TestSuite",
  "log",
  "_method_"
], function(
  TestSuite,
  log,
  _method_
) {
  return TestSuite.extend({
    "initialize": function() {
      log("testsuite initialized successfully!");
      var xTestSuite = this;
      xTestSuite.set( "MethodUnderTest", "_method_" );
            
      xTestSuite.test();
    }
  });
});
